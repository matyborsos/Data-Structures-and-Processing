import requests
from bs4 import BeautifulSoup
from wordcloud import WordCloud

# Scrape the top-level page for the top 1000 Github Python-based repositories
url = "https://github.com/trending/python?since=monthly"
response = requests.get(url)
soup = BeautifulSoup(response.text, "html.parser")

# Extract the topics from each repository and concatenate them into a single string
topics = ""
for repo in soup.select("h1 a"):
    repo_url = "https://github.com" + repo["href"]
    repo_response = requests.get(repo_url)
    repo_soup = BeautifulSoup(repo_response.text, "html.parser")
    for topic in repo_soup.select(".topic-tag"):
        topics += topic.text.lower() + " "

# Create a word cloud from the concatenated string of topics
wordcloud = WordCloud(width=800, height=400, background_color="white").generate(topics)

# Display the word cloud
plt.imshow(wordcloud, interpolation="bilinear")
plt.axis("off")
plt.show()

import requests
import pandas as pd

# Define the endpoint URL and parameters
url = 'https://api.github.com/search/repositories?q=created:>2023-01-01+language:python&sort=stars&order=desc&per_page=1000'

# Make the API request
response = requests.get(url)

# Check if the request was successful
if response.status_code != 200:
    print(f'Request failed with status code {response.status_code}')
    exit()

# Parse the response data into a pandas DataFrame
repos = []
for repo in response.json()['items']:
    repos.append({
        'Name': repo['name'],
        'Owner': repo['owner']['login'],
        'URL': repo['html_url'],
        'Description': repo['description'],
        'Language': repo['language'],
        'Stars': repo['stargazers_count'],
        'Forks': repo['forks_count'],
        'Open Issues': repo['open_issues_count'],
        'Created At': repo['created_at'],
        'Updated At': repo['updated_at'],
        'Percentage': repo['watchers_count']
    })
df = pd.DataFrame(repos)
